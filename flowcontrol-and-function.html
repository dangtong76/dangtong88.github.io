4.1.1 if문

if (3 < 10) {
  print("TRUE")
  print("3은 10보다 작습니다.")

}

4.1.4 연습문제

아래 테이블 과 같은 형태로 데이터 프레임을 만들고 grade 컬럼을 추가하세요

(grade는 200이상일때는 A, 100이상일때는 B, 100미만일때는 C)

Table 4.1: 연습문제 테이블
no name pay grade
1 김재진 110 B
2 구영후 150 B
3 김반장 80 C
4 지소윤 210 A 
no <- c(1:4)
name <- c("김재진","구영후","김반장","지소윤")
pay <- c(110,150,80,210)
test_table <- data.frame(no,name,pay)

test_table$grade <- ifelse(test_table$pay >= 200, "A" , ifelse(test_table$pay >= 100, "B",  "C"))

test_table
##   no   name pay grade
## 1  1 김재진 110     B
## 2  2 구영후 150     B
## 3  3 김반장  80     C
## 4  4 지소윤 210     A

4.1.7 while-next 문

짝수만 출력

i <- 0
while(i<=9){
  i <- i+1
  if(i %% 2 !=0){
    next # print() 를 수행하지 않고 while 문의 처음으로 감
  }
  print(i)
}
## [1] 2
## [1] 4
## [1] 6
## [1] 8
## [1] 10

4.2 연산자

나머지 연산

x <- c(1, 10, 13, 3)
x %% 2 # 나머지연산
## [1] 1 0 1 1
x%/% 3 # 정수나누기의 몫
## [1] 0 3 4 1

부등식/부정

x > 3  # 부등식 
## [1] FALSE  TRUE  TRUE FALSE
y <- c(3, 5, 2, 1)
x > y    # 부등식 - 벡터간 비교 
## [1] FALSE  TRUE  TRUE  TRUE
z <- TRUE
!z      # Not
## [1] FALSE

논리 연산

x1 <- c(T, T, F, F)
y1 <- c(F, F, T, F)
x1 | y1  # or 벡터연산 
## [1]  TRUE  TRUE  TRUE FALSE
T || F   # 처음 값만 비교: 단일값 
## [1] TRUE
x1 &  y1  # and 벡터연산 
## [1] FALSE FALSE FALSE FALSE
x1 && y1  # 처음 값만 비교: 단일값
## [1] FALSE
